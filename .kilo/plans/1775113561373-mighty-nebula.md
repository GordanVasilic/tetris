# Comprehensive Plan: Mobile Touch Controls for Tetris Game

## Executive Summary

This document outlines a comprehensive strategy for implementing intuitive, responsive mobile touch controls for the DEVELOTRIX Tetris game. The current implementation relies solely on keyboard input (arrow keys, spacebar), which is unusable on touchscreen devices where the game board occupies the full screen.

---

## 1. Context and Objectives

### 1.1 Problem Statement

The DEVELOTRIX Tetris game currently:
- Uses keyboard-only controls (Arrow keys, Space, P)
- Displays a responsive canvas that can fill mobile screens
- Has no touch interaction handlers for gameplay
- Shows side panels with stats/controls info that disappear on mobile (<768px breakpoint)

**Result:** The game is unplayable on mobile devices without an external keyboard.

### 1.2 Expected Outcomes

| Goal | Description |
|------|-------------|
| **Visibility** | Controls must be accessible and visible at all times during gameplay |
| **Simplicity** | Intuitive interaction requiring minimal learning curve |
| **Responsiveness** | <50ms latency from touch to visual feedback |
| **Non-intrusive** | Minimal obstruction of the game board while maintaining accessibility |

### 1.3 Key Performance Indicators (KPIs)

```
┌─────────────────────────────────────────────────────────────┐
│                    MOBILE CONTROL KPIs                      │
├─────────────────────────────────────────────────────────────┤
```

### 6.3 Gesture Recognition Implementation

```javascript
// Pseudo-code for gesture recognizer

class TouchController {
    constructor(canvas, gameAPI) {
        this.canvas = canvas;
        this.game = gameAPI;
        this.touchPoints = new Map();
        this.gestureHistory = [];
        
        this.initTouchHandlers();
        this.initGestureRecognition();
    }
    
    initTouchHandlers() {
        this.canvas.addEventListener('touchstart', 
            this.onTouchStart.bind(this), { passive: false });
        this.canvas.addEventListener('touchmove', 
            this.onTouchMove.bind(this), { passive: false });
        this.canvas.addEventListener('touchend', 
            this.onTouchEnd.bind(this), { passive: false });
    }
    
    onTouchStart(e) {
        e.preventDefault(); // Prevent scrolling
        
        for (let touch of e.changedTouches) {
            this.touchPoints.set(touch.identifier, {
                x: touch.clientX,
                y: touch.clientY,
                time: performance.now(),
                element: document.elementFromPoint(touch.clientX, touch.clientY)
            });
        }
        
        // Detect double-tap for hard drop
        this.checkDoubleTap();
    }
    
    onTouchMove(e) {
        e.preventDefault();
        
        for (let touch of e.changedTouches) {
            const point = this.touchPoints.get(touch.identifier);
            if (!point) continue;
            
            const dx = touch.clientX - point.x;
            const dy = touch.clientY - point.y;
            
            // Track movement for gesture recognition
            this.gestureHistory.push({
                id: touch.identifier,
                dx, dy,
                time: performance.now()
            });
            
            point.x = touch.clientX;
            point.y = touch.clientY;
        }
    }
    
    onTouchEnd(e) {
        e.preventDefault();
        
        for (let touch of e.changedTouches) {
            this.processGesture(touch.identifier);
            this.touchPoints.delete(touch.identifier);
        }
    }
    
    processGesture(touchId) {
        // Get gesture history for this touch
        const gestures = this.gestureHistory.filter(
            g => g.id === touchId && 
                 performance.now() - g.time < 500
        );
        
        if (gestures.length === 0) return;
        
        // Calculate net movement
        const netDx = gestures.reduce((sum, g) => sum + g.dx, 0);
        const netDy = gestures.reduce((sum, g) => sum + g.dy, 0);
        
        // Recognize swipe direction
        if (Math.abs(netDx) > Math.abs(netDy)) {
            // Horizontal swipe
            if (netDx > SWIPE_THRESHOLD) this.game.moveRight();
            else if (netDx < -SWIPE_THRESHOLD) this.game.moveLeft();
        } else {
            // Vertical swipe
            if (netDy > SWIPE_THRESHOLD) this.game.softDrop();
        }
        
        // Clear processed gestures
        this.gestureHistory = this.gestureHistory.filter(
            g => g.id !== touchId
        );
    }
    
    checkDoubleTap() {
        const now = performance.now();
        const recentTaps = this.tapHistory.filter(t => now - t < 300);
        
        if (recentTaps.length >= 1) {
            // Double tap detected → Hard drop
            this.game.hardDrop();
            this.triggerHaptic('hard_drop');
        }
        
        this.tapHistory.push(now);
        this.tapHistory = this.tapHistory.filter(t => now - t < 300);
    }
}
```

### Main Touch Controller Module

```javascript
/**
 * MobileTouchController - Handles all touch input for mobile gameplay
 */
class MobileTouchController {
    constructor(game, options = {}) {
        this.game = game;
        this.options = {
            swipeThreshold: 30,
            tapDuration: 200,
            doubleTapWindow: 300,
            enableHaptics: true,
            ...options
        };
        
        this.state = {
            touches: new Map(),
            lastTapTime: 0,
            lastTapPos: null,
            isDragging: false
        };
        
        this.init();
    }
    
    init() {
        // Get game canvas or container
        this.element = document.getElementById('mainCanvas');
        
        // Touch event listeners
        this.element.addEventListener('touchstart', 
            this.handleTouchStart.bind(this), { passive: false });
        this.element.addEventListener('touchmove', 
            this.handleTouchMove.bind(this), { passive: false });
        this.element.addEventListener('touchend', 
            this.handleTouchEnd.bind(this), { passive: false });
        this.element.addEventListener('touchcancel', 
            this.handleTouchCancel.bind(this), { passive: false });
        
        // Prevent default browser behaviors
        this.element.style.touchAction = 'none';
    }
    
    handleTouchStart(e) {
        e.preventDefault();
        
        const now = performance.now();
        
        for (const touch of e.changedTouches) {
            // Check for double-tap (hard drop)
            if (this.isDoubleTap(touch, now)) {
                this.game.hardDrop();
                this.vibrate('hard_drop');
                continue;
            }
            
            // Single tap = rotate
            if (!this.state.isDragging) {
                this.game.rotatePiece();
                this.vibrate('rotate');
            }
            
            // Track touch for swipe detection
            this.state.touches.set(touch.identifier, {
                x: touch.clientX,
                y: touch.clientY,
                time: now
            });
        }
        
        this.state.lastTapTime = now;
    }
    
    handleTouchMove(e) {
        e.preventDefault();
        
        for (const touch of e.changedTouches) {
            const original = this.state.touches.get(touch.identifier);
            if (!original) continue;
            
            const dx = touch.clientX - original.x;
            const dy = touch.clientY - original.y;
            
            // Detect swipe direction
            if (Math.abs(dx) > Math.abs(dy)) {
                // Horizontal movement
                if (dx > this.options.swipeThreshold) {
                    this.game.movePiece(1, 0);
                    this.vibrate('move');
                    original.x = touch.clientX; // Reset for continuous swipe
                } else if (dx < -this.options.swipeThreshold) {
                    this.game.movePiece(-1, 0);
                    this.vibrate('move');
                    original.x = touch.clientX;
                }
            } else {
                // Vertical movement (soft drop)
                if (dy > this.options.swipeThreshold) {
                    const moved = this.game.movePiece(0, 1);
                    if (moved) {
                        this.vibrate('soft_drop');
                        this.game.score += 1;
                    }
                    original.y = touch.clientY;
                }
            }
        }
    }
    
    handleTouchEnd(e) {
        e.preventDefault();
        for (const touch of e.changedTouches) {
            this.state.touches.delete(touch.identifier);
        }
    }
    
    handleTouchCancel(e) {
        this.state.touches.clear();
    }
    
    isDoubleTap(touch, now) {
        const timeDiff = now - this.state.lastTapTime;
        if (timeDiff > this.options.doubleTapWindow) return false;
        
        if (!this.state.lastTapPos) return false;
        
        const dist = Math.hypot(
            touch.clientX - this.state.lastTapPos.x,
            touch.clientY - this.state.lastTapPos.y
        );
        
        return dist < 50; // Within 50px of previous tap
    }
    
    vibrate(pattern) {
        if (!this.options.enableHaptics || !navigator.vibrate) return;
        
        const patterns = {
            move: 5,
            rotate: 10,
            soft_drop: 8,
            hard_drop: [30, 50]
        };
        
        navigator.vibrate(patterns[pattern] || 10);
    }
}

// Usage:

    swipeThreshold: 25,
    enableHaptics: true
});
```

### Button Control Module

```javascript
/**
 * OnScreenControls - Renders and handles on-screen button controls
 */
class OnScreenControls {
    constructor(game, containerId) {
        this.game = game;
        this.container = document.getElementById(containerId);
        this.collapsed = false;
        
        this.render();
        this.setupAutoHide();
    }
    
    render() {
        this.container.innerHTML = `
            <div class="control-rail" id="controlRail">
                <button class="control-btn btn-left" aria-label="Move Left">◄</button>
                <button class="control-btn btn-right" aria-label="Move Right">►</button>
                <button class="control-btn btn-rotate" aria-label="Rotate">↻</button>
                <button class="control-btn btn-drop" aria-label="Hard Drop">⤓</button>
            </div>
        `;
        
        // Bind button handlers
        document.querySelector('.btn-left').addEventListener('touchstart',
            (e) => { e.preventDefault(); this.game.movePiece(-1, 0); });
        
        document.querySelector('.btn-right').addEventListener('touchstart',
            (e) => { e.preventDefault(); this.game.movePiece(1, 0); });
        
        document.querySelector('.btn-rotate').addEventListener('touchstart',
            (e) => { e.preventDefault(); this.game.rotatePiece(); });
        
        document.querySelector('.btn-drop').addEventListener('touchstart',
            (e) => { e.preventDefault(); this.game.hardDrop(); });
        
        // Show rail on any touch
        document.addEventListener('touchstart', () => this.show());
    }
    
    setupAutoHide() {
        let hideTimer;
        
        const hideAfterDelay = () => {
            hideTimer = setTimeout(() => {
                if (this.game.isRunning && !this.game.isPaused) {
                    this.hide();
                }
            }, 3000);
        };
        
        document.addEventListener('touchstart', () => {
            clearTimeout(hideTimer);
            this.show();
            hideAfterDelay();
        });
    }
    
    show() {
        this.container.classList.remove('collapsed');
        this.collapsed = false;
    }
    
    hide() {
        this.container.classList.add('collapsed');
        this.collapsed = true;
    }
    
    toggle() {
        this.collapsed ? this.show() : this.hide();
    }
}
```
